CREATE DATABASE HijabOnlineStore;
USE HijabOnlineStore;

-- Supplier Table
CREATE TABLE Supplier (
    Supplier_ID VARCHAR(10) PRIMARY KEY,
    Supp_Name VARCHAR(100) NOT NULL,
    Contact_Info VARCHAR(100),
    Supp_Address VARCHAR(100)
);

-- Material Table
CREATE TABLE Material (
    Material_ID VARCHAR(100) PRIMARY KEY,
    Material_Type VARCHAR(50) NOT NULL,
    Material_Description TEXT,
    Cost_Per_Unit FLOAT(2),
    Supplier_ID VARCHAR(10),
    FOREIGN KEY (Supplier_ID) REFERENCES Supplier(Supplier_ID)
);

-- Chiffon Table
CREATE TABLE Chiffon (
    Material_ID VARCHAR(10) PRIMARY KEY,
    chiffon_type VARCHAR(50),
    FOREIGN KEY (Material_ID) REFERENCES Material(Material_ID)
);

-- Cotton Table
CREATE TABLE Cotton (
    Material_ID VARCHAR(10) PRIMARY KEY,
    cotton_type VARCHAR(50),
    FOREIGN KEY (Material_ID) REFERENCES Material(Material_ID)
);

-- Silk Table
CREATE TABLE Silk (
    Material_ID VARCHAR(10) PRIMARY KEY,
    silk_type VARCHAR(50),
    FOREIGN KEY (Material_ID) REFERENCES Material(Material_ID)
);

-- Product Table
CREATE TABLE Product (
    Product_ID VARCHAR(10) PRIMARY KEY,
    Prod_Name VARCHAR(100) NOT NULL,
    Prod_Description TEXT,
    Prod_Price FLOAT(2) NOT NULL,
    Prod_Material VARCHAR(50),
    Stock_Quantity INT,
    Supplier_ID VARCHAR(10),
    FOREIGN KEY (Supplier_ID) REFERENCES Supplier(Supplier_ID)
);

-- Bawal Table
CREATE TABLE Bawal (
    Product_ID VARCHAR(10) PRIMARY KEY,
    Bawal_Size VARCHAR(50),
    FOREIGN KEY (Product_ID) REFERENCES Product(Product_ID)
);

-- Shawl Table
CREATE TABLE Shawl (
    Product_ID VARCHAR(10) PRIMARY KEY,
    Shawl_Size VARCHAR(50),
    FOREIGN KEY (Product_ID) REFERENCES Product(Product_ID)
);

-- Inventory Table
CREATE TABLE Inventory (
    Inventory_ID VARCHAR(10) PRIMARY KEY,
    Product_ID VARCHAR(10),
    Inv_Quantity INT,
    Last_Updated DATE,
    FOREIGN KEY (Product_ID) REFERENCES Product(Product_ID)
);

-- Customer Table
CREATE TABLE Customer (
    Customer_ID VARCHAR(10) PRIMARY KEY,
    Cust_Name VARCHAR(100) NOT NULL,
    Cust_Email VARCHAR(100) UNIQUE,
    Cust_Address TEXT,
    Phone_Number VARCHAR(15)
);

-- Order Table
CREATE TABLE HijabOrder (
    Order_ID VARCHAR(10) PRIMARY KEY,
    Order_Date DATE NOT NULL,
    Total_Amount FLOAT (2) NOT NULL,
    Shipping_ID VARCHAR(10),
    Customer_ID VARCHAR(10),
    Product_ID VARCHAR(10),
    FOREIGN KEY (Product_ID) REFERENCES Product(Product_ID),
    FOREIGN KEY (Customer_ID) REFERENCES Customer(Customer_ID)
);

-- Shipping Table
CREATE TABLE Shipping (
    Shipping_ID VARCHAR(10) PRIMARY KEY,
    Order_ID VARCHAR(10),
    Tracking_Number VARCHAR(50),
    Shipping_Date DATE,
    Delivery_Date DATE,
    Shipping_Status VARCHAR(50),
    FOREIGN KEY (Order_ID) REFERENCES HijabOrder (Order_ID)
);

-- Payment Table
CREATE TABLE Payment (
    Payment_ID VARCHAR(10) PRIMARY KEY,
    Order_ID VARCHAR(10),
    Payment_Date DATE NOT NULL,
    Payment_Amount FLOAT(2) NOT NULL,
    Payment_Method VARCHAR(50),
    Payment_Status VARCHAR(50),
    FOREIGN KEY (Order_ID) REFERENCES HijabOrder (Order_ID)
);

-- Return Table
CREATE TABLE OrderReturn (
    Return_ID VARCHAR(10) PRIMARY KEY,
    Order_ID VARCHAR(10),
    Order_Date DATE,
    Reason TEXT,
    Refund_Amount FLOAT(2),
    Refund_Status VARCHAR(50),
    FOREIGN KEY (Order_ID) REFERENCES HijabOrder(Order_ID)
);

-- Review Table
CREATE TABLE Review (
    Review_ID VARCHAR(10) PRIMARY KEY,
    Product_ID VARCHAR(10),
    Customer_ID VARCHAR(10),
    Review_Comment TEXT,
    Rating INT,
    FOREIGN KEY (Product_ID) REFERENCES Product(Product_ID),
    FOREIGN KEY (Customer_ID) REFERENCES Customer(Customer_ID)
);

SELECT 
    Customer.Cust_Name 
    AS Customer_Name,
    Customer.Cust_Email 
    AS Email,
    HijabOrder.Order_ID 
    AS Order_ID,
    HijabOrder.Order_Date 
    AS Order_Date,
    Product.Prod_Name
    AS Product_Name,
    Product.Prod_Price 
    AS Product_Price,
    Shipping.Shipping_Status
    AS Shipping_Status
FROM 
    Customer
JOIN 
    HijabOrder 
    ON Customer.Customer_ID = HijabOrder.Customer_ID
JOIN 
    Product 
    ON HijabOrder.Product_ID = Product.Product_ID
JOIN 
    Shipping 
    ON HijabOrder.Order_ID = Shipping.Order_ID
WHERE 
    Shipping.Shipping_Status = 'Delivered';
